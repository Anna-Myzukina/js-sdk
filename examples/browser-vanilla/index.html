Timekit JS SDK vanilla example

This is the most minimal boilerplate demo of timekit.js used in the browser. Fire up your console (CMD+ALT+I) and use the SDK by hitting methods on the "timekit" global variable, e.g.

timekit.configure({
  app: 'myNewApp',
  inputTimestampFormat: 'l h:i a Y-m-d'
});

timekit.auth('some_user@gmail.com', 'somepassword').then(function(response) {
  console.log(response);
});

timekit.findTime(['some_user@gmail.com', 'some_other_user@gmail.com'], '3 days', '30 minutes').then(function(response) {
  console.log(response);
});
